6. Suricata Rules Management

# Force rule update
sudo /opt/codered/bin/update-rules.sh

# Check last update
cat /var/log/codered/last-rule-update.log

# View rule count
grep -c "^alert\|^drop" /etc/suricata/rules/*.rules

# Add custom rules
sudo vi /etc/suricata/rules/local.rules

# Example:
alert http any any -> any any (msg:"CUSTOM Suspicious UA"; http.user_agent; content:"evil-bot"; sid:9000001; rev:1;)

# Reload
sudo systemctl restart suricata

10. Troubleshooting

No Traffic Being Captured

# Check monitor interfaces are promisc
ip -d link show eth1 | grep -i promisc

# Test packet capture
sudo tcpdump -i eth1 -c 10

# If no packets: SPAN/mirror is misconfigured on the switch

Zeek Not Running

sudo /opt/zeek/bin/zeekctl status
sudo /opt/zeek/bin/zeekctl diag    # show crash info
sudo /opt/zeek/bin/zeekctl deploy  # redeploy

Logs Not Reaching CodeRed AI

# Check Filebeat
sudo systemctl status filebeat
sudo journalctl -u filebeat -n 50

# Test connectivity
curl -sk https://<codered-ai-ip>:9200

Customer Locked Out

# SSH as your admin user, reset password:
# Customer manages their own user password via standard 'passwd' command

Rule Update Failing

cat /var/log/codered/rule-update.log
# Common causes: no internet, DNS failure, disk full
curl -sI https://rules.emergingthreats.net

12. Multi-Sensor Fleet Management

# Create sensor list
cat > ~/sensors.txt <<EOF
10.50.1.100
10.50.2.100
192.168.1.100
EOF

# Check all sensors
for ip in $(cat ~/sensors.txt); do
    echo "=== $ip ==="
    ssh admin@$ip "hostname; uptime -p; cat /opt/codered/VERSION; pgrep -c zeek; pgrep -c suricata"
    echo ""
done

# Push rule update to all
for ip in $(cat ~/sensors.txt); do
    ssh admin@$ip "sudo /opt/codered/bin/update-rules.sh" &
done
wait

# Push config update to all
for ip in $(cat ~/sensors.txt); do
    ssh admin@$ip "sudo /opt/codered/bin/codered-update.sh" &
done
wait
